```bash

for f in sorted(pathlib.Path("charness-artifacts/cautilus").glob("*/finding.md")):
    t=f.read_text(); d=f.parent; m=[]
    if not any(x in t for x in SRC): m.append("SRC")
    if not any(x in t for x in VERD): m.append("VERD")
    if not any(x in t for x in INT): m.append("INT")
    if not any((d/e).is_file() for e in EV): m.append("EVID")
    if m: print(d.name, m)
PY
```
